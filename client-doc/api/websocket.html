WebSocket连接 ​

连接地址： ws://{host}:{port}/websocket

连接参数：

javascript

// 建立WebSocket连接
const ws = new WebSocket('ws://localhost:9401/websocket');

// 发送认证消息
ws.onopen = () => {
  ws.send(JSON.stringify({
    type: 'AUTH',
    token: localStorage.getItem('token')
  }));
};

// 接收消息
ws.onmessage = (event) => {
  const message = JSON.parse(event.data);
  handleMessage(message);
};
